require("dotenv").config(); const express = require("express"); const cors = require("cors"); const axios = require("axios"); const app = express(); app.use(cors()); app.use(express.json()); const PORT = process.env.PORT || 3000; const FUENTES = [ { nombre: "Salud Integral (SIG)", endpoint: "https://api-sintegral.soemgroup.com.ar", loginPath: "/login/", usuario: process.env.USUARIO_SIG, clave: process.env.CLAVE_SIG, }, { nombre: "SIG Clínica (CEMEI)", endpoint: "https://api-cemei.soemgroup.com.ar", loginPath: "/login/", usuario: process.env.USUARIO_CEMEI, clave: process.env.CLAVE_CEMEI, }, { nombre: "Neoform", endpoint: "https://api-neo.soemgroup.com.ar", loginPath: "/loginUserApi/", usuario: process.env.USUARIO_NEO, clave: process.env.CLAVE_NEO, }, ]; const tokenCache = {}; const TOKEN_TTL_MS = 5 * 60 * 1000; async function obtenerToken(fuente) { const cache = tokenCache[fuente.endpoint]; if (cache && Date.now() - cache.obtenidoEn < TOKEN_TTL_MS) return cache.token; const respuesta = await axios.post(`${fuente.endpoint}${fuente.loginPath}`, { username: fuente.usuario, password: fuente.clave, }); const token = respuesta.data.tokenAuth; tokenCache[fuente.endpoint] = { token, obtenidoEn: Date.now() }; return token; } function convertirFecha(fechaTexto) { if (!fechaTexto) return null; const [dia, mes, anio] = fechaTexto.split("-"); return `${anio}-${mes}-${dia}`; } async function obtenerDetallePagos(fuente, fechaDesde, fechaHasta) { const token = await obtenerToken(fuente); const respuesta = await axios.post( `${fuente.endpoint}/consultar_detalle_pagos/`, { fecha_desde: fechaDesde, fecha_hasta: fechaHasta }, { headers: { "X-Authorization-token": token } } ); const lista = respuesta.data.listadoDetPagos || []; return lista.map((item) => ({ idPago: item.id_pago, proveedor: item.proveedor, banco: item.banco, importe: parseFloat(item.importe), concepto: item.concepto, cobrado: item.cobrado, fechaCobro: convertirFecha(item.fecha_cobro), rubro: item.rubro, origen: fuente.nombre, })); } app.get("/api/pagos", async (req, res) => { const { desde, hasta } = req.query; if (!desde) return res.status(400).json({ error: "El parámetro 'desde' es obligatorio (yyyy-mm-dd)" }); // allSettled: si una API falla, las otras igual devuelven datos const resultados = await Promise.allSettled( FUENTES.map((f) => obtenerDetallePagos(f, desde, hasta || desde)) ); const errores = []; const combinado = []; resultados.forEach((r, i) => { if (r.status === "fulfilled") { combinado.push(...r.value); } else { const msg = `${FUENTES[i].nombre}: ${r.reason?.message || "error desconocido"}`; errores.push(msg); console.error("Error en fuente", msg); } }); res.json({ total: combinado.length, importeTotal: combinado.reduce((a, p) => a + (p.importe || 0), 0), pagos: combinado, errores: errores.length > 0 ? errores : undefined, }); }); app.get("/", (req, res) => res.send("Backend financiero activo. Usá /api/pagos?desde=YYYY-MM-DD&hasta=YYYY-MM-DD")); app.listen(PORT, () => console.log(`Servidor corriendo en puerto ${PORT}`));
