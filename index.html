const PRIMER_MES_DISPONIBLE = "2026-01"; const FUENTES = [ { codigo: "SIG", nombre: "Salud Integral (SIG)", endpoint: "https://api-sintegral.soemgroup.com.ar", loginPath: "/login/", usuario: process.env.USUARIO_SIG, clave: process.env.CLAVE_SIG, cobranzasPath: "/consultar_cobranzas/", cobranzasKey: "registrosCobranzas", }, { codigo: "CEMEI", nombre: "SIG Clínica (CEMEI)", endpoint: "https://api-cemei.soemgroup.com.ar", loginPath: "/login/", usuario: process.env.USUARIO_CEMEI, clave: process.env.CLAVE_CEMEI, facturacionPath: "/listFacturacion/", facturacionKey: "registrosFacturas", }, { codigo: "NEO", nombre: "Neoform", endpoint: "https://api-neo.soemgroup.com.ar", loginPath: "/loginUserApi/", usuario: process.env.USUARIO_NEO, clave: process.env.CLAVE_NEO, facturacionPath: "/listFacturacion/", facturacionKey: "registrosFacturas", }, ]; // ---------- TOKEN CACHE ---------- const tokenCache = {}; const TOKEN_TTL_MS = 5 * 60 * 1000; async function obtenerToken(fuente) { const cache = tokenCache[fuente.endpoint]; if (cache && Date.now() - cache.obtenidoEn < TOKEN_TTL_MS) return cache.token; const respuesta = await axios.post(`${fuente.endpoint}${fuente.loginPath}`, { username: fuente.usuario, password: fuente.clave, }); const token = respuesta.data.tokenAuth; if (!token) throw new Error(`Login sin token para ${fuente.nombre}`); tokenCache[fuente.endpoint] = { token, obtenidoEn: Date.now() }; return token; } // ---------- HELPERS ---------- function convertirFecha(fechaTexto) { // dd-MM-yyyy → yyyy-MM-dd if (!fechaTexto) return null; const [dia, mes, anio] = fechaTexto.split("-"); return `${anio}-${mes}-${dia}`; } function ultimoDiaDeMes(anioMes) { const [anio, mes] = anioMes.split("-").map(Number); const dia = new Date(Date.UTC(anio, mes, 0)).getUTCDate(); return `${anioMes}-${String(dia).padStart(2, "0")}`; } function sumarMeses(anioMes, cantidad) { let [anio, mes] = anioMes.split("-").map(Number); mes += cantidad; while (mes > 12) { mes -= 12; anio += 1; } while (mes < 1) { mes += 12; anio -= 1; } return `${anio}-${String(mes).padStart(2, "0")}`; } // ---------- EGRESOS: /api/pagos ---------- async function obtenerDetallePagos(fuente, fechaDesde, fechaHasta) { const token = await obtenerToken(fuente); const respuesta = await axios.post( `${fuente.endpoint}/consultar_detalle_pagos/`, { fecha_desde: fechaDesde, fecha_hasta: fechaHasta }, { headers: { "X-Authorization-token": token } } ); const lista = respuesta.data.listadoDetPagos || []; return lista.map((item) => { const rawCaja = (item.empresa_caja || "").toUpperCase().trim(); let unidad = item.empresa_caja || null; if (rawCaja === "CONVENIO EXTERNO") unidad = "Farmacia"; else if (rawCaja === "CEMEI ODONTOLOGIA") unidad = "CEMEI Odontología"; else if (rawCaja === "CEMEI SRL") unidad = "CEMEI SRL"; return { idPago: item.id_pago, proveedor: item.proveedor, banco: item.banco, importe: parseFloat(item.importe), concepto: item.concepto, cobrado: item.cobrado, fechaCobro: convertirFecha(item.fecha_cobro), rubro: item.rubro, origen: fuente.nombre, unidad: unidad, }; }); } app.get("/api/pagos", async (req, res) => { const { desde, hasta } = req.query; if (!desde) return res.status(400).json({ error: "El parámetro 'desde' es obligatorio (yyyy-mm-dd)" }); const resultados = await Promise.allSettled( FUENTES.map((f) => obtenerDetallePagos(f, desde, hasta || desde)) ); const errores = []; const combinado = []; resultados.forEach((r, i) => { if (r.status === "fulfilled") combinado.push(...r.value); else { const msg = `${FUENTES[i].nombre}: ${r.reason?.message || "error"}`; errores.push(msg); console.error("Error en fuente", msg); } }); res.json({ total: combinado.length, importeTotal: combinado.reduce((a, p) => a + (p.importe || 0), 0), pagos: combinado, errores: errores.length > 0 ? errores : undefined, }); }); // ---------- INGRESOS: /api/cobros ---------- // SIG → /consultar_cobranzas/ directo con el rango exacto // CEMEI/NEO → /listFacturacion/ ampliado, filtrado por fecha_cobro en el período function normalizarCobranzaSIG(item, fuente) { const monto = parseFloat(item.importe) || 0; return { empresa: fuente.nombre, fechaCobro: item.fecha_cobrado || item.fecha_emision || null, cliente: item.cliente || item.obra_social || "Sin cliente", concepto: item.concepto_factura || item.concepto || "", comprobante: item.comprobante || item.numero_comprobante || null, formaPago: item.tipo_cobranza || item.banco || "", monto, }; } function normalizarFacturacionCobro(item, fuente) { const monto = parseFloat(item.monto) || 0; const cobradoFlag = ["si", "sí", "1"].includes(String(item.cobrado || "").toLowerCase()); if (!cobradoFlag) return null; return { empresa: fuente.nombre, fechaCobro: item.fecha_cobro || item.fecha || null, cliente: item.cliente || "Sin cliente", concepto: item.concepto || "", comprobante: item.comprobante || null, formaPago: item.forma_pago || "", monto, }; } async function obtenerCobranzasSIG(fuente, desde, hasta) { const token = await obtenerToken(fuente); const respuesta = await axios.post( `${fuente.endpoint}${fuente.cobranzasPath}`, { fecha_desde: desde, fecha_hasta: hasta }, { headers: { "X-Authorization-token": token } } );
